15def singleton(cls):
16 """!
17 Décorateur pour implémenter le patron Singleton.
18
19 **Parameters**
20 - **cls**: La classe à décorer.
21
22 **Returns**
23 - Une fonction wrapper retournant l'instance unique de la classe.
24 """
25 instance = [None]
26 def wrapper(*args, **kwargs):
27 if instance[0] is None:
28 instance[0] = cls(*args, **kwargs)
29 return instance[0]
30 return wrapper

248 def concorde(self, pattern, context_size=30):
249 """!
250 Génère une concordance pour un motif donné dans le corpus.
251
252 **Parameters**
253 - **pattern**: Motif (expression régulière) à rechercher.
254 - **context_size**: Taille du contexte à extraire autour du motif.
255
256 **Returns**
257 - DataFrame avec colonnes "contexte gauche", "motif trouvé", "contexte droit".
258 """
259 if self._full_text is None:
260 self._build_full_text()
261 text = self._full_text
262
263 matches = re.finditer(pattern, text, re.IGNORECASE)
264
265 results = []
266
267 for match in matches:
268 start_index = match.start()
269 end_index = match.end()
270 motif = match.group()
271
272 # Extraire le contexte gauche
273 # Début du contexte = max(0, start_index - taille du contexte)
274 context_gauche_start = max(0, start_index - context_size)
275 context_gauche = text[context_gauche_start:start_index].strip()
276
277 # Extraire le contexte droit
278 # Fin du contexte = min(longueur totale, end_index + taille du contexte)
279 context_droit_end = min(len(text), end_index + context_size)
280 context_droit = text[end_index:context_droit_end].strip()
281
282 results.append({
283 "contexte gauche": context_gauche,
284 "motif trouvé": motif,
285 "contexte droit": context_droit
286 })
287
288 if not results:
289 print(f"Aucune occurrence du motif '{pattern}' trouvée.")
290 return pd.DataFrame(columns=["contexte gauche", "motif trouvé", "contexte droit"])
291
292 concordancier = pd.DataFrame(results)
293 return concordancier

295 def nettoyer_texte(self, text):
296 """!
297 Nettoie le texte en appliquant plusieurs transformations.
298
299 **Parameters**
300 - **text**: Texte brut à nettoyer.
301
302 **Returns**
303 - Texte nettoyé (minuscules, caractères non alphabétiques retirés, espaces normalisés).
304 """
305 text = text.lower()
306
307 text = text.replace('\n', ' ')
308
309 # [^a-z] signifie "tout ce qui n'est pas une lettre minuscule"
310 text = re.sub(r'[^a-z\s]+', ' ', text)
311
312 # Remplacement des espaces multiples par un seul espace
313 text = re.sub(r'\s+', ' ', text).strip()
314
315 return text

317 def stats(self, n=10):
318 """!
319 Calcule et affiche des statistiques globales et textuelles du corpus.
320
321 **Parameters**
322 - **n**: Nombre de mots les plus fréquents à afficher.
323
324 **Returns**
325 - DataFrame trié par fréquence décroissante (TF).
326 """
327 print(f"\n--- Statistiques du corpus '{self.nom}' ---")
328 print(f"Nombre de documents: {len(self.documents)}")
329 print(f"Nombre d'auteurs: {len(self.authors)}")
330 print("\nStats textuelles :")
331
332 # Dictionnaire pour stocker les occurrences globales (Term Frequency)
333 word_counts = {}
334
335 # Dictionnaire pour stocker les documents contenant chaque mot (Document Frequency)
336 doc_word_set = {}
337
338 all_words = []
339
340 for doc in self.documents.values():
341 text_nettoye = self.nettoyer_texte(doc.get_texte())
342
343 mots_du_doc = text_nettoye.split(' ')
344
345 mots_uniques_du_doc = set(mots_du_doc)
346
347 for mot in mots_uniques_du_doc:
348 if mot:
349 doc_word_set[mot] = doc_word_set.get(mot, 0) + 1
350
351 # Ajout des mots pour le calcul de la fréquence totale (TF)
352 all_words.extend(mots_du_doc)
353
354
355 # Utilisation d'un dictionnaire pour compter la fréquence (plus rapide que boucler à la main)
356 word_counts = Counter([word for word in all_words if word])
357
358 df_freq = pd.DataFrame(word_counts.items(), columns=['Mot', 'Term Frequency (TF)'])
359
360 df_freq['Document Frequency (DF)'] = df_freq['Mot'].apply(lambda mot: doc_word_set.get(mot, 0))
361
362 print(f"1. Nombre de mots différents dans le corpus (Vocabulaire): {len(word_counts)} mots.")
363
364 print(f"\n2. {n} mots les plus fréquents (Term Frequency, TF) :")
365
366 # Tri par Term Frequency et affichage des n premiers
367 df_freq_sorted = df_freq.sort_values(by='Term Frequency (TF)', ascending=False)
368
369 print(df_freq_sorted.head(n))
370
371 return df_freq_sorted
