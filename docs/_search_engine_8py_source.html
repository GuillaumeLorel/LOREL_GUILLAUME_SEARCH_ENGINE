26 def __init__(self, corpus):
27 """!
28 Constructeur qui lance toutes les étapes d'indexation.

46 def _build_vocab(self):
47 """!
48 Construction du vocabulaire à partir du corpus.
49 """
50 mots_uniques = set()
51 docs = list(self.corpus.get_documents().values())
52
53 for doc in docs:
54 texte = self.corpus.nettoyer_texte(doc.get_texte())
55 for mot in texte.split(' '):
56 if mot:
57 mots_uniques.add(mot)
58
59 liste_mots = sorted(list(mots_uniques))
60
61 self.vocab = {}
62 for i, mot in enumerate(liste_mots):
63 self.vocab[mot] = {
64 'id': i,
65 'doc_count': 0
66 }
67 print(f"-> Vocabulaire créé : {len(self.vocab)} mots.")

69 def _build_tf_matrix(self):
70 """!
71 Construction de la matrice Documents x Mots (TF).
72 """
73 rows = []
74 cols = []
75 data = []
76
77 docs = list(self.corpus.get_documents().values())
78
79 for index_doc, doc in enumerate(docs):
80 texte = self.corpus.nettoyer_texte(doc.get_texte())
81 mots_doc = [m for m in texte.split(' ') if m]
82
83 compte_local = {}
84 for mot in mots_doc:
85 compte_local[mot] = compte_local.get(mot, 0) + 1
86
87 # Remplissage des listes pour la matrice sparse
88 for mot, count in compte_local.items():
89 if mot in self.vocab:
90 index_mot = self.vocab[mot]['id']
91
92 rows.append(index_doc)
93 cols.append(index_mot)
94 data.append(count)
95
96 self.vocab[mot]['doc_count'] += 1
97
98 n_vocab = len(self.vocab)
99 self.mat_TF = csr_matrix((data, (rows, cols)), shape=(self.N_docs, n_vocab))

101 def _build_tfidf_matrix(self):
102 """!
103 Construction de la matrice TF-IDF.
104 """
105 idf_list = []
106 mots_tries = sorted(self.vocab.keys())
107
108 for mot in mots_tries:
109 df = self.vocab[mot]['doc_count']
110 # Formule IDF
111 if df > 0:
112 val_idf = math.log(self.N_docs / df)
113 else:
114 val_idf = 0
115 idf_list.append(val_idf)
116
117 # Multiplication : Matrice TF * Diagonale des IDF
118 diag_idf = diags(idf_list)
119
120 self.mat_TF_IDF = self.mat_TF.dot(diag_idf)

122 def search(self, query, n_results=10):
123 """!
124 Recherche des documents les plus pertinents pour une requête.
125
126 **Parameters**
127 - **query**: La requête utilisateur.
128 - **n_results**: Nombre de documents à retourner.
129
130 **Returns**
131 - Un DataFrame avec les résultats triés par score décroissant.
132
133 **Notes**
134 - Retourne un DataFrame vide si aucun terme de la requête n'est dans le vocabulaire.
135 """
136 query_clean = self.corpus.nettoyer_texte(query)
137 mots_query = [m for m in query_clean.split(' ') if m]
138
139 query_vec = np.zeros(len(self.vocab))
140
141 for mot in mots_query:
142 if mot in self.vocab:
143 idx = self.vocab[mot]['id']
144 query_vec[idx] += 1
145
146 scores = []
147 docs = list(self.corpus.get_documents().values())
148
149 # Norme du vecteur requête ||B||
150 norm_query = np.linalg.norm(query_vec)
151
152 if norm_query == 0:
153 return pd.DataFrame()
154
155 # Boucle sur chaque document
156 for i in tqdm(range(self.N_docs), desc="Recherche", leave=False):
157 doc_vec = self.mat_TF_IDF.getrow(i).toarray()[0]
158
159 # Produit scalaire A . B
160 dot_product = np.dot(doc_vec, query_vec)
161
162 # Norme du document ||A||
163 norm_doc = np.linalg.norm(doc_vec)
164
165 if norm_doc > 0:
166 sim = dot_product / (norm_doc * norm_query)
167 else:
168 sim = 0
169
170 scores.append(sim)
171
172 indices_tries = np.argsort(scores)[::-1]
173
174 resultats = []
175 for i in indices_tries[:n_results]:
176 if scores[i] > 0:
177 doc_obj = docs[i]
178 resultats.append({
179 "Document": doc_obj.get_titre(),
180 "Score": round(scores[i], 4),
181 "Auteur": doc_obj.get_auteur(),
182 "Date": doc_obj.get_date(),
183 "URL": doc_obj.get_url(),
184 "Type": doc_obj.getType()
185 })
186
187 return pd.DataFrame(resultats)
